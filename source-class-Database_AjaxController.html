736     public function upload()
737     {
738         if ($this->_helper->Identity()) {
739             //Check if images path directory is writable
740             if (!is_writable(IMAGE_PATH)) {
741                 throw new Pas_Exception_NotAuthorised('The images directory is not writable', 500);
742             }
743             // Create the imagedir path
744             $imagedir = IMAGE_PATH . '/' . $this->_helper->Identity()->username;
745 
746             //Check if a directory and if not make directory
747             if (!is_dir($imagedir)) {
748                 mkdir($imagedir, 0775, true);
749             }
750             //Check if the personal image directory is writable
751             if (!is_writable($imagedir)) {
752                 throw new Pas_Exception_NotAuthorised('The user image directory is not writable', 500);
753             }
754 
755             // Get images and do the magic
756             $adapter = new Zend_File_Transfer_Adapter_Http();
757             $adapter->setDestination($imagedir);
758             $adapter->setOptions(array('useByteString' => false));
759             // Only allow good image files!
760             $adapter->addValidator('Extension', false, 'jpg, tiff');
761             $adapter->addValidator('NotExists', false, array($imagedir));
762             $files = $adapter->getFileInfo();
763 
764             // Create an array for the images
765             $images = array();
766 
767             // Loop through the submitted files
768             foreach ($files as $file => $info) {
769 
770 
771                 // file uploaded & is valid
772 //                if (!$adapter->isUploaded($file)) continue;
773 //                if (!$adapter->isValid($file)) continue;
774 
775                 // Clean up the image name for crappy characters
776                 $filename = pathinfo($adapter->getFileName($file));
777                 // Instantiate the renamer
778                 $reNamer = new Pas_Image_Rename();
779                 // Clean the filename
780                 $cleaned = $reNamer->strip($filename['filename'], $filename['extension']);
781                 // Rename the file
782                 $adapter->addFilter('rename', $cleaned);
783                 // receive the files into the user directory
784                 $adapter->receive($file); // this has to be on top
785                 if(!$adapter->hasErrors()){
786                     // Create the object for reuse
787                     $image = new stdClass();
788                     $image->cleaned = $cleaned;
789                     $image->basename = $filename['basename'];
790                     $image->extension = $filename['extension'];
791                     $image->thumbnailUrl = $this->createThumbnailUrl($adapter->getFileName($file, false));
792                     $image->deleteUrl = $this->_createUrl($adapter->getFileName($file, false));
793                     $image->path = $adapter->getFileName($file);
794                     $image->name = $adapter->getFileName($file, false);
795                     $image->size = $adapter->getFileSize($file);
796                     $image->mimetype = $adapter->getMimeType($file);
797                     // The secure ID stuff for linking images
798                     $image->secuid = $this->_helper->GenerateSecuID();
799                     // Get the image dims
800                     $imagesize = getimagesize($adapter->getFileName($file));
801                     $image->width = $imagesize[0];
802                     $image->height = $imagesize[1];
803                     $params = $this->getAllParams();
804                     $image->findID = $params['findID'];
805                     // Create the raw image url
806                     $image->url = $this->_createUrl($adapter->getFileName($file, false));
807                     $image->deleteType = 'DELETE';
808                     $images[] = $image;
809                     $slides = new Slides();
810                     $insert = $slides->addAndResize($images);
811                     $this->view->data = $images;
812                     $this->_helper->solrUpdater->update('images', (int)$insert);
813                     $this->_helper->solrUpdater->update('objects', $params['findID'], 'artefacts');
814                 } else {
815                     $image = new stdClass();
816                     $image->error = $adapter->getErrors();
817                     $images[] = $image;
818                     $this->view->data = $images;
819                 }
820             }
821 
822         } else {
823             throw new Pas_Exception_NotAuthorised('Your account does not seem enabled to do this', 500);
824         }
825     }
