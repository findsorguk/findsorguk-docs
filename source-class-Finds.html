 115     public function generateFindId()
 116     {
 117         $institution = $this->getInstitution();
 118         if (!is_null($institution)) {
 119             list($usec, $sec) = explode(" ", microtime());
 120             $suffix = strtoupper(substr(dechex($sec), 3) . dechex(round($usec * 15)));
 121             $findid = $institution . '-' . $suffix;
 122             return $findid;
 123         } else {
 124             throw new Pas_Exception_Group('Institution missing', 500);
