405      */
406     public function preProcess()
407     {
408         $tier = 0;
409         $tileGroupNumber = 0;
410         $numberOfTiles = 0;
411         if ($this->getDebug() == true) {
412             Zend_Debug::dump($this->_scaleInfo, 'SCALEINFO');
413         }
414         foreach ($this->_scaleInfo as $width_height) {
415             list($width, $height) = $width_height;
416             #       cycle through columns, then rows
417             $row = 0;
418             $column = 0;
419             $ul_x = 0;
420             $ul_y = 0;
421             $lr_x = 0;
422             $lr_y = 0;
423             while (!(($lr_x == $width) && ($lr_y == $height))) {
424 
425                 $tileFileName = $this->getTileFileName($tier, $column, $row);
426                 if ($this->getDebug()) {
427                     Zend_Debug::dump($tileFileName, 'tile mappings filename' . __LINE__);
428                 }
429                 $tileContainerName = $this->getNewTileContainerName($tileGroupNumber);
430                 if ($this->getDebug()) {
431                     Zend_Debug::dump($tileContainerName, 'TILE container name');
432                 }
433                 if ($numberOfTiles == 0) {
434                     $this->createTileContainer($tileContainerName);
435                 } elseif ($numberOfTiles % $this->getTileSize() == 0) {
436                     $tileGroupNumber++;
437                     $tileContainerName = $this->getNewTileContainerName($tileGroupNumber);
438                     if ($this->getDebug()) {
439                         Zend_Debug::dump($tileContainerName, 'TILE container name');
440                     }
441                     $this->createTileContainer($tileContainerName);
442                 }
443                 $this->_tileGroupMappings[$tileFileName] = $tileContainerName;
444                 if ($this->getDebug()) {
445                     Zend_Debug::dump($this->_tileGroupMappings, 'tile mappings ' . __LINE__);
446                 }
447                 $numberOfTiles++;
448 
449                 if ($this->getDebug()) {
450                     Zend_Debug::dump($numberOfTiles, 'number of tiles');
451                 }
452                 # for the next tile, set lower right cropping point
453                 if ($ul_x + $this->getTileSize() < $width) {
454                     $lr_x = $ul_x + $this->getTileSize();
455                 } else {
456                     $lr_x = $width;
457                 }
458 
459                 if ($ul_y + $this->getTileSize() < $height) {
460                     $lr_y = $ul_y + $this->getTileSize();
461                 } else {
462                     $lr_y = $height;
463                 }
464 
465                 # for the next tile, set upper left cropping point
466                 if ($lr_x == $width) {
467                     $ul_x = 0;
468                     $ul_y = $lr_y;
469                     $column = 0;
470                     $row++;
471                 } else {
472                     $ul_x = $lr_x;
473                     $column++;
474                 }
475             }
476             $tier++;

897 
898 function rm($fileglob)
899 {
900     if (is_string($fileglob)) {
901         if (is_file($fileglob)) {
902             return unlink($fileglob);
903         } else if (is_dir($fileglob)) {
904             $ok = rm("$fileglob/*");
905             if (!$ok) {
906                 return false;
907             }
908             return rmdir($fileglob);
909         } else {
910             $matching = glob($fileglob);
911             if ($matching === false) {
912                 trigger_error(sprintf('No files match supplied glob %s', $fileglob), E_USER_WARNING);
913                 return false;
914             }
915             $rcs = array_map('rm', $matching);
916             if (in_array(false, $rcs)) {
917                 return false;
918             }
919         }
920     } else if (is_array($fileglob)) {
921         $rcs = array_map('rm', $fileglob);
922         if (in_array(false, $rcs)) {
923             return false;
924         }
925     } else {
926         trigger_error('Param #1 must be filename or glob pattern, or array of filenames or glob patterns', E_USER_ERROR);
927         return false;
928     }
929     return true;
930 }
