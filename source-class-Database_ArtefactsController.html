252     /** Add an object
253      * @access public
254      * @return void
255      * @todo slim down action, move logic for adding to finds.php model
256      */
257     public function addAction()
258     {
259         $user = $this->_helper->identity->getPerson();
260         if ((is_null($user->peopleID) && is_null($user->canRecord))
261             || (!is_null($user->peopleID) && is_null($user->canRecord))
262         ) {
263             $this->redirect('/error/accountproblem');
264         }
265         $last = $this->getParam('copy');
266         $this->view->secuid = $this->secuid();
267         $form = $this->getFindForm();
268         $form->submit->setLabel('Save record');
269         $form->old_findID->setValue($this->FindUid());
270         $form->secuid->setValue($this->secuid());
271         if (isset($user->peopleID)) {
272             $form->recorderID->setValue($user->peopleID);
273             $form->recordername->setValue($user->fullname);
274             $form->identifier1ID->setValue($user->peopleID);
275             $form->idBy->setValue($user->fullname);
276         }
277         if (in_array($user->role, $this->_restricted)) {
278             $form->finderID->setValue($user->peopleID);
279             $form->removeDisplayGroup('discoverers');
280             $form->removeElement('finder');
281             $form->removeElement('secondfinder');
282             $form->removeElement('idBy');
283             $form->recordername->setAttrib('disabled', true);
284             $form->removeElement('id2by');
285         }
286         $this->view->form = $form;
287         if ($last == 'last') {
288             $findData = $this->getFinds()->getLastRecord($this->getIdentityForForms());
289             foreach ($findData as $findDataFlat) {
290                 $form->populate($findDataFlat);
291                 if (isset($user->peopleID)) {
292                     $form->recorderID->setValue($user->peopleID);
293                     $form->recordername->setValue($user->fullname);
294                 }
295             }
296         }
297         if ($this->getRequest()->isPost()) {
298             if ($form->isValid($this->_request->getPost())) {
299                 $insertData = $form->getValues();
300                 $insert = $this->getFinds()->addFind($insertData);
301                 if ($insert != 'error') {
302                     $this->_helper->solrUpdater->update('objects', $insert, 'artefacts');
303                     $this->redirect(self::REDIRECT . 'record/id/' . $insert);
304                     $this->getFlash()->addMessage('Record created!');
305                 } else { // If there is a database error, repopulate form so users don't lose their work
306                     $this->getFlash()->addMessage('Database error. Please try submitting again or contact support.');
307                     $form->populate($this->_request->getPost());
308                 }
309             } else {
310                 $this->getFlash()->addMessage('Please check and correct errors!');
311                 $form->populate($this->_request->getPost());
312             }
313         }

588     private function _getAdviser($objecttype, $broadperiod)
589     {
590         $config = $this->_helper->config();
591         $this->_romancoinsadviser = $config->findsadviser->romancoins;
592         $this->_romancoinsadviseremail = $config->findsadviser->romcoins->email;
593 
594         $this->_medievalcoinsadviser = $config->findsadviser->medievalcoins;
595         $this->_medievalcoinsadviseremail = $config->findsadviser->medcoins->email;
596 
597         $this->_romanobjects = $config->findsadviser->romanobjects;
598         $this->_romanobjectsemail = $config->findsadviser->romobjects->email;
599 
600         $this->_medievalobjects = $config->findsadviser->medievalobjects;
601         $this->_medievalobjectsemail = $config->findsadviser->medobjects->email;
602 
603         $this->_postmedievalobjects = $config->findsadviser->postmedievalobjects;
604         $this->_postmedievalobjectsemail = $config->findsadviser->postmedobjects->email;
605 
606         $this->_earlymedievalobjects = $config->findsadviser->earlymedievalobjects;
607         $this->_earlymedievalobjectsemail = $config->findsadviser->earlymedobjects->email;
608 
609         $this->_catchall = $config->findsadviser->default;
610         $this->_catchallemail = $config->findsadviser->def->email;
611 
612         switch ($objecttype) {
613 
614             case (in_array($objecttype, $this->_coinarray)
615                 && in_array($broadperiod, $this->_periodRomIA)):
616                 $adviserdetails = $this->_romancoinsadviser;
617                 $adviseremail = $this->_romancoinsadviseremail;
618                 break;
619 
620             case (in_array($objecttype, $this->_coinarray)
621                 && in_array($broadperiod, $this->_earlyMed)):
622                 $adviserdetails = $this->_medievalcoinsadviser;
623                 $adviseremail = $this->_medievalcoinsadviseremail;
624                 break;
625 
626             case (in_array($objecttype, $this->_coinarray)
627                 && in_array($broadperiod, $this->_medieval)):
628                 $adviserdetails = $this->_medievalcoinsadviser;
629                 $adviseremail = $this->_medievalcoinsadviseremail;
630                 break;
631 
632             case (in_array($objecttype, $this->_coinarray)
633                 && in_array($broadperiod, $this->_postMed)):
634                 $adviserdetails = $this->_medievalcoinsadviser;
635                 $adviseremail = $this->_medievalcoinsadviseremail;
636                 break;
637 
638             case (!in_array($objecttype, $this->_coinarray)
639                 && in_array($broadperiod, $this->_periodRomPrehist)):
640                 $adviserdetails = $this->_romanobjects;
641                 $adviseremail = $this->_romanobjectsemail;
642                 break;
643 
644             case (!in_array($objecttype, $this->_coinarray)
645                 && in_array($broadperiod, $this->_postMed)):
646                 $adviserdetails = $this->_postmedievalobjects;
647                 $adviseremail = $this->_postmedievalobjectsemail;
648                 break;
649 
650             case (!in_array($objecttype, $this->_coinarray)
651                 && in_array($broadperiod, $this->_medieval)):
652                 $adviserdetails = $this->_medievalobjects;
653                 $adviseremail = $this->_medievalobjectsemail;
654                 break;
655 
656             case (!in_array($objecttype, $this->_coinarray)
657                 && in_array($broadperiod, $this->_earlyMed)):
658                 $adviserdetails = $this->_earlymedievalobjects;
659                 $adviseremail = $this->_earlymedievalobjectsemail;
660                 break;
661 
662             default:
663                 $adviserdetails = $this->_catchall;
664                 $adviseremail = $this->_catchallemail;
665                 break;
666         }
667         $combine = new Pas_ArrayFunctions();
668         $people = $combine->combine($adviserdetails->toArray(), $adviseremail->toArray());
669         $sendto = array();
670         foreach ($people as $k => $v) {
671             $sendto[] = array('email' => $v, 'name' => $k);
672         }
673         return $sendto;
674     }
