 642     /** Filter parameters being passed to search
 643      * @access public
 644      * @param array $params
 645      * @return array
 646      */
 647     public function filterParams(array $params)
 648     {
 649         if (array_key_exists('created', $params)) {
 650             $created = $params['created'];
 651             $queryDateA = $created . "T00:00:00.001Z";
 652             $queryDateB = $created . "T23:59:59.99Z";
 653             $params['created'] = $queryDateA . ' TO ' . $queryDateB;
 654         }
 655         if (array_key_exists('updated', $params)) {
 656             $updated = $params['updated'];
 657             $queryDateA = $updated . "T00:00:00.001Z";
 658             $queryDateB = $updated . "T23:59:59.99Z";
 659             $params['updated'] = $queryDateA . ' TO ' . $queryDateB;
 660         }
 661         if (array_key_exists('createdBefore', $params) && array_key_exists('createdAfter', $params)) {
 662             $queryDateA = $params['createdAfter'] . "T00:00:00.001Z";
 663             $queryDateB = $params['createdBefore'] . "T23:59:59.99Z";
