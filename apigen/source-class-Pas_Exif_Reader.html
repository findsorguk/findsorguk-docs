 540                 case self::M_SOF6:

 543                 case self::M_SOF10:
 544                 case self::M_SOF11:
 545                 case self::M_SOF13:
 546                 case self::M_SOF14:
 547                 case self::M_SOF15:

 570         }
 571         $i++;
 572         if($exitAll == 1)  break;
 573             //if($tmpTestLevel == 2)  break;
 574         }
 575         fclose($fp);
 576         $this->_newFile = 0;
 577     }
